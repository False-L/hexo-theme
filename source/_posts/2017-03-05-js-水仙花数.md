# 求1000以内的水仙花数

```javascript
function t(num) {
var num = num;
  var a = 0;
  var b = 0;
  var c = 0;
  var t1 = 0;
  for (var i = 100; i < num; i++)
  {
    a = i % 10;
    b = parseInt(i/10)  % 10;
    c = parseInt(i/100) % 10;
    t1 = Math.pow(a,3) + Math.pow(b,3) + Math.pow(c,3);
    if (i == t1) {
      alert(num + "以内的水仙花数"+ i);
    }
  }
}
t(1000);
```
