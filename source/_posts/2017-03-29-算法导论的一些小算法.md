## 直接插入排序
 - 升序
```javascript
var a=[5,2,4,6,1,3,10,3,8];
console.log(a);
for (var j=1;j<a.length;j++){
        var key=a[j];
        var i=j-1;
        while(a[i]>key){
            a[i+1]=a[i];
            i=i-1;
            a[i+1]=key;
        }
}
console.log("a"+"["+a+"]"+a[0]);
```

## 选择排序
```javascript
var a=[5,2,4,6,1,3,10,3,8];
console.log(a);
for(var j=0;j<a.length;j++)
{
    var min=j;
    var i=j+1;
    while(a[i]&&i<a.length+1){
       if(a[min]>a[i]){
          min=i;
       }
        i=i+1;
    }
    var key=a[j];
    a[j]=a[min];
    a[min]=key;
}
console.log("a"+"["+a+"]"+"a[0]="+a[0]);
```
